# Apply deployments
kubectl apply -f blue_deployment.yaml
kubectl apply -f green_deployment.yaml

# Apply service
kubectl apply -f kubeservice.yaml

# Scale deployment
kubectl scale deployment django-messaging-deployment --replicas=3

# Check pods
kubectl get pods

# Check logs
kubectl logs <pod-name>

# Monitor resources
kubectl top pods

# Apply ingress
kubectl apply -f ingress.yaml

# Describe ingress
kubectl describe ingress django-messaging-ingress

# Port forwarding (if not using ingress)
kubectl port-forward service/django-messaging-service 8000:8000

# Access via curl (if using ingress and DNS is configured)
curl http://messaging.local/
